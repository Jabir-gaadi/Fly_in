# THE GRAND SWARM TEST - 50 DRONES
# This map tests the load balancing algorithm by offering 3 distinct main routes:
# 1. The Highway: Short distance, high node capacity, but extremely tight links (capacity=2)
# 2. The Residential Area: Wide links (capacity=5) but zones are restricted (takes 2 turns)
# 3. The Industrial Park: Very wide links and high capacity, but the longest route out of the way.

nb_drones: 50

start_hub: start 0 0 [color=green max_drones=50]
end_hub: end 10 0 [color=cyan max_drones=50]

hub: hwy1 2 2 [color=yellow max_drones=25]
hub: hwy2 5 2 [color=yellow max_drones=25]

# === ROUTE 2: RESIDENTIAL (Restricted Zones - 2 turns) ===
hub: res1 3 -2 [zone=restricted color=orange max_drones=20]

# === ROUTE 3: INDUSTRIAL PARK (Wide open but long) ===
